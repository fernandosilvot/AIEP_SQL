CREATE PROCEDURE CALCULA_TOTAL @IDENTIFICADOR AS INT 
AS 
BEGIN 

select Sociedad_Operadora, SUM(UTM) from sancionesAplicadas INNER JOIN DATOS_SOCIEDAD ON DATOS_SOCIEDAD.ID_SOCIEDAD = sancionesAplicadas.SOCIEDAD_ID WHERE ID = @IDENTIFICADOR
GROUP BY Sociedad_Operadora

END

ALTER PROCEDURE PROMEDIO @N1 INT, @N2 INT, @N3 INT 
AS 
BEGIN 


DECLARE @SUMA INT 
DECLARE @PROMEDIO INT 

 SET @SUMA = @N1 + @N2 + @N3
 SET @PROMEDIO = @SUMA/3

 SELECT @PROMEDIO AS PROMEDIO 

 IF @PROMEDIO < 40
	PRINT 'REPROBADO'
ELSE 
	PRINT'APROBADO'


END
